8.1 Shebang e Estrutura de um Script

#!/bin/bash
#===============================================================================
# Nome: backup.sh
# Descricao: Faz backup de diretorios importantes
# Autor: Seu Nome
# Data: 2024-06-14
# Uso: ./backup.sh [diretorio]
#===============================================================================

set -euo pipefail
# -e: sai se algum comando falhar
# -u: erro ao usar variavel nao definida
# -o pipefail: pipeline falha se qualquer comando falhar

# Variaveis
BACKUP_DIR="/backup"
DATA=$(date +%Y-%m-%d)
LOG="$BACKUP_DIR/backup_$DATA.log"

# Funcoes
log() {
    echo "[$(date '+%Y-%m-%d %H:%M:%S')] $1" | tee -a "$LOG"
}

# Codigo principal
log "Iniciando backup..."
# ...
log "Backup concluido!"

8.2 Variaveis

8.2.1 Declaracao e Uso

#!/bin/bash

# Declaracao (SEM espacos ao redor de =)
NOME="Joao"
IDADE=25
PI=3.14159

# Uso (com $)
echo "Ola, $NOME!"
echo "Voce tem ${IDADE} anos"

# Variavel nao definida (com -u, gera erro)
# echo "$NAO_EXISTE"  # ERRO!

# Valor padrao
echo "${NAO_EXISTE:-padrao}"    # imprime "padrao"
echo "${NAO_EXISTE:=padrao}"    # define e imprime "padrao"
echo "${EXISTE:?"nao definida"}" # erro se nao definida

# Remover prefixo/sufixo
ARQUIVO="documento.txt"
echo "${ARQUIVO%.txt}"     # documento

echo "${ARQUIVO#doc}"      # umento.txt

# Substituicao
TEXTO="hello world hello"
echo "${TEXTO/hello/HI}"   # HI world hello (primeira)
echo "${TEXTO//hello/HI}"  # HI world HI (todas)

# Maiusculas/minusculas
echo "${TEXTO^^}"          # HELLO WORLD HELLO
echo "${TEXTO,,}"          # hello world hello

# Comprimento
echo "${#TEXTO}"           # 17

# Substring
TEXTO="abcdef"
echo "${TEXTO:2}"          # cdef (a partir do indice 2)
echo "${TEXTO:2:3}"        # cde (3 caracteres a partir do 2)

8.2.3 Arrays

#!/bin/bash

# Declarar array
FRUTAS=("maca" "banana" "laranja")

# Acessar elementos
echo "${FRUTAS[0]}"     # maca
echo "${FRUTAS[1]}"     # banana

# Todos os elementos
echo "${FRUTAS[@]}"     # maca banana laranja
echo "${FRUTAS[*]}"     # maca banana laranja

# Numero de elementos
echo "${#FRUTAS[@]}"    # 3

# Adicionar elemento
FRUTAS+=("uva")

# Iterar
for fruta in "${FRUTAS[@]}"; do
    echo "Fruta: $fruta"
done

# Iterar com indice
for i in "${!FRUTAS[@]}"; do
    echo "$i: ${FRUTAS[$i]}"
done

# Remover elemento
unset FRUTAS[1]

# Array associativo (bash 4+)
declare -A USUARIOS
USUARIOS[joao]=25
USUARIOS[maria]=30
echo "${USUARIOS[joao]}"  # 25

# Todas as chaves
echo "${!USUARIOS[@]}"    # joao maria

8.3 Condicionais — if, case

8.3.1 if

#!/bin/bash

# Basico
if [ "$IDADE" -ge 18 ]; then
    echo "Maior de idade"
fi

# if-else
if [ "$IDADE" -ge 18 ]; then
    echo "Maior de idade"
else
    echo "Menor de idade"
fi

# if-elif-else
if [ "$NOTA" -ge 90 ]; then
    echo "A"
elif [ "$NOTA" -ge 80 ]; then
    echo "B"
elif [ "$NOTA" -ge 70 ]; then
    echo "C"
else
    echo "D"
fi

# Testes com && e ||
[ -f "$ARQUIVO" ] && echo "Existe" || echo "Nao existe"

# Teste inverso
if ! [ -f "$ARQUIVO" ]; then
    echo "Arquivo nao existe"
fi

8.4 Loops — for, while, until

8.4.1 for

#!/bin/bash

# Iterar sobre lista
for i in 1 2 3 4 5; do
    echo "Numero: $i"
done

# Range (bash 4+)
for i in {1..10}; do
    echo "$i"
done

# Range com passo
for i in {1..10..2}; do
    echo "$i"  # 1, 3, 5, 7, 9
 done

# Iterar sobre array
for fruta in "${FRUTAS[@]}"; do
    echo "$fruta"
done

# Iterar sobre arquivos
for arquivo in *.txt; do
    echo "Processando: $arquivo"
done

# C-style for
for ((i=0; i<10; i++)); do
    echo "$i"
done

# Ler linhas de arquivo
while IFS= read -r linha; do
    echo "$linha"
done < arquivo.txt

# Iterar sobre saida de comando
for usuario in $(cut -d: -f1 /etc/passwd); do
    echo "Usuario: $usuario"
done

# Iterar com find (seguro para nomes com espacos)
while IFS= read -r -d '' arquivo; do
    echo "Arquivo: $arquivo"
done < <(find . -name "*.txt" -print0)

8.4.2 while

#!/bin/bash

# Loop basico
CONTADOR=0
while [ "$CONTADOR" -lt 10 ]; do
    echo "$CONTADOR"
    ((CONTADOR++))
done

# Ler arquivo linha por linha
while IFS= read -r linha; do
    echo "Linha: $linha"
done < arquivo.txt

# Loop infinito
while true; do
    echo "Executando..."
    sleep 5
done

# Loop com condicao de arquivo
while [ -f "/tmp/lock" ]; do
    echo "Aguardando..."
    sleep 1
done

# Loop com read (input do usuario)
while true; do
    read -p "Digite 'sair' para encerrar: " resposta
    [ "$resposta" = "sair" ] && break
done

8.4.3 until

#!/bin/bash

# Executa ATE a condicao ser verdadeira
CONTADOR=0
until [ "$CONTADOR" -ge 10 ]; do
    echo "$CONTADOR"
    ((CONTADOR++))
done

# Esperar arquivo aparecer
until [ -f "/tmp/pronto" ]; do
    echo "Aguardando arquivo..."
    sleep 1
done

8.4.4 Controle de Loop

#!/bin/bash

# break: sai do loop
for i in {1..10}; do
    [ "$i" -eq 5 ] && break
    echo "$i"
done
# Sai no 5, imprime 1-4

# continue: pula para proxima iteracao
for i in {1..10}; do
    [ "$i" -eq 5 ] && continue
    echo "$i"
done
# Pula o 5, imprime 1-4, 6-10

# break N: sai de N niveis de loop
for i in {1..3}; do
    for j in {1..3}; do
        [ "$j" -eq 2 ] && break 2
        echo "$i-$j"
    done
done
# Sai dos dois loops

8.5 Funcoes

#!/bin/bash

# Declaracao basica
minha_funcao() {
    echo "Ola da funcao!"
}

# Ou
function minha_funcao {
    echo "Ola da funcao!"
}

# Chamada
minha_funcao

# Com parametros
soma() {
    local a=$1
    local b=$2
    echo $((a + b))
}

RESULTADO=$(soma 5 3)
echo "Resultado: $RESULTADO"

# Retorno (codigo de saida)
verificar_arquivo() {
    if [ -f "$1" ]; then
        return 0  # sucesso
    else
        return 1  # falha
    fi
}

if verificar_arquivo "/etc/passwd"; then
    echo "Arquivo existe"
else
    echo "Arquivo nao existe"
fi

# Variaveis locais
funcao() {
    local VAR="local"
    echo "$VAR"
}
funcao
echo "$VAR"  # vazio (nao definido no escopo global)

# Funcao recursiva
fatorial() {
    local n=$1
    if [ "$n" -le 1 ]; then
        echo 1
    else
        local prev=$(fatorial $((n - 1)))
        echo $((n * prev))
    fi
}

echo "5! = $(fatorial 5)"

# Passar array para funcao
processar_array() {
    local arr=("$@")
    for item in "${arr[@]}"; do
        echo "Item: $item"
    done
}

processar_array "${FRUTAS[@]}"

8.6 Parametros e Argumentos

#!/bin/bash

# Nome do script
SCRIPT="$0"

# Argumentos posicionais
PRIMEIRO="$1"
SEGUNDO="$2"
TERCEIRO="$3"

# Todos os argumentos
TODOS="$@"
TODOS_STRING="$*"

# Numero de argumentos
NUMERO="$#"

# Ultimo argumento
ULTIMO="${!#}"

# Shift (remove primeiro argumento)
shift
echo "Agora \$1 eh: $1"

# Shift N
shift 2

# getopts (parse de opcoes)
while getopts "a:b:vh" opt; do
    case $opt in
        a)
            ARG_A="$OPTARG"
            ;;
        b)
            ARG_B="$OPTARG"
            ;;
        v)
            VERBOSE=1
            ;;
        h)
            echo "Uso: $0 -a valor -b valor [-v] [-h]"
            exit 0
            ;;
        \?)
            echo "Opcao invalida: -$OPTARG" >&2
            exit 1
            ;;
    esac
done
shift $((OPTIND - 1))

# Uso: ./script.sh -a valor1 -b valor2 -v arquivo.txt

# Argumentos restantes apos getopts
echo "Arquivos: $@"
